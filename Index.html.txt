        const calcModal = document.getElementById('calculator-modal');

        function calculateLatePenalty() {
            const mins = parseInt(minutesInput.value) || 0;
            let fine = 0;
            let absence = "None";
            let note = "";

            if (mins <= 0) {
                fine = 0;
                absence = "None";
                note = "On time! No action needed.";
            } else if (mins <= 10) {
                fine = 0;
                absence = "None";
                note = "Within 10 minutes grace period. No fine applies.";
            } else if (mins <= 30) {
                // $100 per minute after the first 10 minutes
                fine = (mins - 10) * 100;
                if (fine > 500) fine = 500;
                absence = "None";
                note = `Fined $100 for each minute late past grace period. Current fine is $${fine}.`;
            } else if (mins < 60) {
                fine = 500;
                absence = "0.5 Absence (Half Absence)";
                note = "Late for more than 30 minutes. Counted as one-half absence and maximum late fine of $500.";
            } else {
                fine = 500;
                absence = "1.0 Absence (Full Absence)";
                note = "Late for more than 1 hour. Handled as a full absence with a $500 penalty.";
            }

            fineText.innerText = `$${fine}`;
            absenceText.innerText = absence;
            noteText.innerText = "* " + note;
        }
